  CRÍTICO #1 — Dashboard completo accesible sin login
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

PROBLEMA:
  La ruta raíz "/" carga el dashboard completo sin requerir
  autenticación. Tabs de Broker, Portafolio, Señales, Bitácora
  y Alertas están visibles para cualquier visitante anónimo.
  NextAuth está configurado en el código pero no se protege
  la ruta principal en el middleware.

FIX — middleware.ts:
  Agregar "/" a las rutas protegidas con redirect a /login:

  export async function middleware(request: NextRequest) {
    const { pathname } = request.nextUrl;

    // Proteger el dashboard principal
    const isDashboard =
      pathname === '/' ||
      pathname.startsWith('/dashboard');

    if (isDashboard) {
      const token = await getToken({
        req: request,
        secret: process.env.NEXTAUTH_SECRET,
      });

      if (!token) {
        const loginUrl = new URL('/login', request.url);
        loginUrl.searchParams.set('callbackUrl', pathname);
        return NextResponse.redirect(loginUrl);
      }
    }

    // ... resto del middleware existente
  }

  // Actualizar el matcher:
  export const config = {
    matcher: ['/', '/dashboard/:path*', '/api/:path*'],
  };

FIX — Crear página de login si no existe:
